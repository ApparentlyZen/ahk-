-- --- FONCTION DE TRANSFORMATION ---
local function toggleMinimize()
    isMinimized = not isMinimized
    if isMinimized then
        for _, item in ipairs(contentElements) do
            item.Visible = false
        end
        titleLabel.Visible = false
        close.Visible = false
        minimize.Visible = false
        
        uicorner.CornerRadius = UDim.new(1, 0) -- Mode Gros Rond 45x45
        reopenText.Visible = true
        
        frame:TweenSize(UDim2.new(0, 45, 0, 45), Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.25, true)
    else
        uicorner.CornerRadius = UDim.new(0, 14)
        reopenText.Visible = false
        
        frame:TweenSize(UDim2.new(0, 220, 0, 155), Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.25, true, function()
            titleLabel.Visible = true
            close.Visible = true
            minimize.Visible = true
            for _, item in ipairs(contentElements) do
                item.Visible = true
            end
        end)
    end
end

-- --- LE CERVEAU UNIQUE DU DRAG ET DU TAP ---
local dragging = false
local dragInput
local dragStart = Vector3.new()
local startPos = UDim2.new()

local startTouchTime = 0
local hasMovedMuch = false

local function onInputBegan(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
        if uis:GetFocusedTextBox() then return end
        
        dragging = true
        hasMovedMuch = false
        dragStart = input.Position
        startPos = frame.Position
        startTouchTime = os.clock()
        
        local connection
        connection = input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
                connection:Disconnect()
                
                -- SI ON EST RÉDUIT : On vérifie si c'est un Tap rapide sans grand mouvement
                if isMinimized and not hasMovedMuch and (os.clock() - startTouchTime) < 0.25 then
                    toggleMinimize()
                end
            end
        end)
    end
end

-- Clic classique du bouton "-" quand le menu est OUVERT
minimize.MouseButton1Click:Connect(function()
    if not isMinimized then
        toggleMinimize()
    end
end)

frame.InputBegan:Connect(onInputBegan)

frame.InputChanged:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        dragInput = input
    end
end)

uis.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        -- Si le doigt bouge de plus de 5 pixels, c'est officiellement un DRAG
        if delta.Magnitude > 5 then
            hasMovedMuch = true
        end
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- --- Logique Infaillible et Invisible ---
local lagBusy = false 
local function dropFPS() 
    if not FEATURE_ENABLED or lagBusy then return end 
    lagBusy = true 
    
    settings().Network.IncomingReplicationLag = 1000 
    
    local stop = os.clock() + LAG_DURATION 
    local interval = 1 / DROP_TO_FPS 
    
    local con 
    con = rs.RenderStepped:Connect(function() 
        if os.clock() > stop then 
            if con then con:Disconnect() end 
            return 
        end 
        local now = os.clock() 
        while os.clock() - now < interval do end 
    end) 
    
    task.wait(LAG_DURATION) 
    settings().Network.IncomingReplicationLag = 0 
    lagBusy = false 
end 

-- Détecteur automatique en arrière-plan (Silencieux)
local function connectAnimationWatcher() 
    local char = lp.Character or lp.CharacterAdded:Wait() 
    local hum = char:WaitForChild("Humanoid") 
    
    if hum:FindFirstChild("__CUPIDS_AUTOMATIC") then 
        hum:FindFirstChild("__CUPIDS_AUTOMATIC"):Destroy() 
    end 
    
    local flag = Instance.new("BoolValue") 
    flag.Name = "__CUPIDS_AUTOMATIC" 
    flag.Parent = hum 
    
    hum.AnimationPlayed:Connect(function(anim) 
        local animId = tostring(anim.Animation.AnimationId) 
        local animName = anim.Name or ""
        
        if animId:find(SANGUINE_ANIMATION_ID) or animId:find(GHOUL_ANIMATION_ID) or animName == GHOUL_ANIMATION_NAME or animId:find(DIAMOND_ANIMATION_ID) then 
            dropFPS() 
        end
    end) 
end 

-- Initialisation
lp.CharacterAdded:Connect(function() 
    task.wait(1) 
    connectAnimationWatcher() 
end) 
if lp.Character then connectAnimationWatcher() end
